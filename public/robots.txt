import { MetadataRoute } from 'next';

/**
 * FULL SEO ROBOTS CONFIGURATION WITH CRAWL DELAY
 * Location: src/app/robots.ts (or app/robots.ts)
 * Remember to delete public/robots.txt so this file can take over.
 */

export default function robots(): MetadataRoute.Robots {
  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL || "https://aps.com.pk";
  const base = siteUrl.replace(/\/$/, "");

  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: [
          '/admin',
          '/api',
          '/checkout/success',
          '/cart',
          '/account',
          '/_next',
          '/static',
        ],
        // @ts-ignore - crawlDelay is supported by Bing/Yandex but not in standard Next.js types
        crawlDelay: 10,
      },
      {
        userAgent: 'Googlebot',
        allow: '/',
        disallow: [
          '/admin',
          '/api',
          '/checkout',
          '/account',
        ],
      },
    ],
    sitemap: `${base}/sitemap.xml`,
    host: base,
  };
}
